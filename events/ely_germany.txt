ely_germany.2 = {
	type = country_event
	placement = ROOT

	title = ely_germany.2.t
	desc = ely_germany.2.d
	flavor = ely_germany.2.f

	event_image = {
		video = "gfx/event_pictures/unspecific_military_parade.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		c:AUS = {
			every_scope_state = {
				limit = {
					is_incorporated = yes
				}
				set_variable = AUS_incorporated_state
			}
		}
		if = {
			limit = {
				NOT = { exists = c:HUN }
				exists = s:STATE_CENTRAL_HUNGARY.region_state:AUS
			}
			create_country = {
				tag = HUN
				origin = c:AUS
				state = s:STATE_CENTRAL_HUNGARY.region_state:AUS
			}
		}
		else_if = {
			limit = {
				exists = c:HUN
				exists = s:STATE_CENTRAL_HUNGARY.region_state:AUS
			}
			s:STATE_CENTRAL_HUNGARY.region_state:AUS = { set_state_owner = c:HUN }
		}
		else_if = {
			limit = {
				NOT = { exists = c:HUN }
			}
			random_country = {
				limit = {
					is_secessionist = yes
					any_scope_state = {
						state_region = s:STATE_CENTRAL_HUNGARY
					}
				}
				change_tag = HUN
			}
		}
		if = {
			limit = { exists = s:STATE_TRANSDANUBIA.region_state:AUS }
			s:STATE_TRANSDANUBIA.region_state:AUS = { set_state_owner = c:HUN }
		}
		if = {
			limit = { exists = s:STATE_CROATIA.region_state:AUS }
			s:STATE_CROATIA.region_state:AUS = { set_state_owner = c:HUN }
		}
		if = {
			limit = { exists = s:STATE_WEST_SLOVAKIA.region_state:AUS }
			s:STATE_WEST_SLOVAKIA.region_state:AUS = { set_state_owner = c:HUN }
		}
		if = {
			limit = { exists = s:STATE_EAST_SLOVAKIA.region_state:AUS }
			s:STATE_EAST_SLOVAKIA.region_state:AUS = { set_state_owner = c:HUN }
		}
		if = {
			limit = { exists = s:STATE_SLAVONIA.region_state:AUS }
			s:STATE_SLAVONIA.region_state:AUS = { set_state_owner = c:HUN }
		}
		c:HUN = {
			activate_law = law_type:law_presidential_republic
			activate_law = law_type:law_universal_suffrage
			activate_law = law_type:law_homesteading
			activate_law = law_type:law_appointed_bureaucrats
			create_character = {
				template = hun_lajos_kossuth_template
			}
			create_character = {
				template = hun_sandor_petofi_template
			}
		}
	}

    option = {
		name = ely_germany.2.a
		default_option = yes
		ai_chance = {
			base = 100
		}
		c:BAV = {
			trigger_event = { id = ely_germany.3 days = 30 }
		}
		c:PRU = {
			ruler = {
				add_modifier = {
					name = german_unifier_ruler
				}
			}

				value = 0.1

			}
