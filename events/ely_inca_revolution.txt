ely_incan_revolution.1 = {
	type = country_event
	placement = root
	title = ely_incan_revolution.1.t
	desc = ely_incan_revolution.1.d
	flavor = ely_incan_revolution.1.f

	duration = 3

	event_image = {
		video = "votp_agitator_speaking"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	trigger = {
		# Triggered by JE
	}

	immediate = {
		if = {
			limit = {
				NOT = { exists = c:INC }
				exists = s:STATE_LA_PAZ.region_state:TWT
			}
			create_country = {
				tag = INC
				origin = c:TWT
				state = s:STATE_LA_PAZ.region_state:TWT
			}
		}
		else_if = {
			limit = {
				exists = c:INC
				exists = s:STATE_LA_PAZ.region_state:TWT
			}
			s:STATE_LA_PAZ.region_state:TWT = { set_state_owner = c:INC }
		}
		else_if = {
			limit = {
				NOT = { exists = c:INC }
			}
			random_country = {
				limit = {
					is_secessionist = yes
					any_scope_state = {
						state_region = s:STATE_LA_PAZ
					}
				}
				change_tag = INC
			}
		}
		if = {
			limit = { exists = s:STATE_AREQUIPA.region_state:TWT }
			s:STATE_AREQUIPA.region_state:TWT = { set_state_owner = c:INC }
		}
		if = {
			limit = { exists = s:STATE_POTOSI.region_state:TWT }
			s:STATE_POTOSI.region_state:TWT = { set_state_owner = c:INC }
		}
		if = {
			limit = { exists = s:STATE_SANTIAGO.region_state:TWT }
			s:STATE_SANTIAGO.region_state:TWT = { set_state_owner = c:INC }
		}
		if = {
			limit = { exists = s:STATE_LOS_RIOS.region_state:TWT }
			s:STATE_LOS_RIOS.region_state:TWT = { set_state_owner = c:INC }
		}
		if = {
			limit = { exists = s:STATE_ARAUCANIA.region_state:TWT }
			s:STATE_ARAUCANIA.region_state:TWT = { set_state_owner = c:INC }
		}
		if = {
			limit = { exists = s:STATE_TARAPACA.region_state:TWT }
			s:STATE_TARAPACA.region_state:TWT = { set_state_owner = c:INC }
		}
		if = {
			limit = { exists = s:STATE_ANTOFAGASTA.region_state:TWT }
			s:STATE_ANTOFAGASTA.region_state:TWT = { set_state_owner = c:INC }
		}
		if = {
			limit = { exists = s:STATE_PASTAZA.region_state:TWT }
			s:STATE_PASTAZA.region_state:TWT = { set_state_owner = c:INC }
		}
		if = {
			limit = { exists = s:STATE_ACRE.region_state:TWT }
			s:STATE_ACRE.region_state:TWT = { set_state_owner = c:INC }
		}
		if = {
			limit = { exists = s:STATE_JUJUY.region_state:TWT }
			s:STATE_JUJUY.region_state:TWT = { set_state_owner = c:INC }
		}
		c:INC = {
			add_modifier = {
				name = incan_revolution_fervor
				months = normal_modifier_time
			}	
			every_scope_state = {
				limit = {
					owner = c:INC
				}
				set_state_type = incorporated
			}
			add_journal_entry = { type = je_win_andean_civilwar }
			ig:ig_landowners = {
				remove_ideology = ideology_paternalistic
				remove_ideology = ideology_patriarchal
				remove_ideology = ideology_stratocratic
				add_ideology = ideology_incan_rich_land
				add_ideology = ideology_incan_rich_rights
				add_ideology = ideology_incan_rich_army
				add_ideology = ideology_incan_rich_economy
			}
			ig:ig_petty_bourgeoisie = {
				remove_ideology = ideology_patriotic
				remove_ideology = ideology_reactionary
				remove_ideology = ideology_meritocratic
				add_ideology = ideology_incan_middle_land
				add_ideology = ideology_incan_middle_rights
				add_ideology = ideology_incan_middle_army
				add_ideology = ideology_incan_middle_economy
			}
			ig:ig_armed_forces = {
				remove_ideology = ideology_jingoist
				remove_ideology = ideology_patriotic
				remove_ideology = ideology_loyalist
				add_ideology = ideology_incan_middle_land
				add_ideology = ideology_incan_middle_rights
				add_ideology = ideology_incan_middle_army
				add_ideology = ideology_incan_middle_economy
			}
			ig:ig_intelligentsia = {
				remove_ideology = ideology_anti_clerical
				remove_ideology = ideology_republican
				remove_ideology = ideology_liberal_modern
				remove_ideology = ideology_anti_slavery
				add_ideology = ideology_incan_middle_land
				add_ideology = ideology_incan_middle_rights
				add_ideology = ideology_incan_middle_army
				add_ideology = ideology_incan_middle_economy
			}
			ig:ig_industrialists = {
				remove_ideology = ideology_laissez_faire
				remove_ideology = ideology_individualist
				remove_ideology = ideology_plutocratic
				add_ideology = ideology_incan_rich_land
				add_ideology = ideology_incan_rich_rights
				add_ideology = ideology_incan_rich_army
				add_ideology = ideology_incan_rich_economy
			}
			ig:ig_devout = {
				remove_ideology = ideology_patriarchal
				remove_ideology = ideology_pious
				remove_ideology = ideology_moralist
				add_ideology = ideology_incan_middle_land
				add_ideology = ideology_incan_middle_rights
				add_ideology = ideology_incan_middle_army
				add_ideology = ideology_incan_middle_economy
			}
			ig:ig_rural_folk = {
				remove_ideology = ideology_particularist
				remove_ideology = ideology_agrarian
				remove_ideology = ideology_isolationist
				add_ideology = ideology_incan_poor_land
				add_ideology = ideology_incan_poor_rights
				add_ideology = ideology_incan_poor_army
				add_ideology = ideology_incan_poor_economy
			}
			ig:ig_trade_unions = {
				remove_ideology = ideology_populist
				remove_ideology = ideology_egalitarian_modern
				remove_ideology = ideology_proletarian
				remove_ideology = ideology_anti_slavery
				add_ideology = ideology_incan_poor_land
				add_ideology = ideology_incan_poor_rights
				add_ideology = ideology_incan_poor_army
				add_ideology = ideology_incan_poor_economy
			}
		}
		c:TWT = {
			add_modifier = {
				name = incan_revolution_sickness
				months = normal_modifier_time
			}	
		}

	}

	option = {
		name = ely_incan_revolution.1.a
		add_loyalists = {
			value = large_radicals
			strata = rich
		}
		create_diplomatic_play = {
			target_country = c:INC
			type = dp_annex_war
			escalation = 81
			target_state = s:STATE_LA_PAZ.region_state:INC
			add_war_goal = {
				holder = c:INC
				type = annex_country
				target_state = s:STATE_LIMA.region_state:TWT
			}
		}
	}
	option = {
		name = ely_incan_revolution.1.b
		default_option = yes
		play_as = c:INC
		create_diplomatic_play = {
			target_country = c:INC
			type = dp_annex_war
			escalation = 81
			target_state = s:STATE_LA_PAZ.region_state:INC
			add_war_goal = {
				holder = c:INC
				type = annex_country

	}
