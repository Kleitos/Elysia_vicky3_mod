ely_infrastructure_event.5 = {
	type = country_event
	event_image = {
		video = "gfx/event_pictures/unspecific_trains.bk2"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/trains"

	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	duration = 3
	placement = scope:florida_scope

	title = ely_infrastructure_event.5.t
	desc = ely_infrastructure_event.5.d
	flavor = ely_infrastructure_event.5.f

	trigger = {
	}
	
	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_FLORIDA
			}
			save_scope_as = florida_scope
		}
	}

	option = {
		name = ely_infrastructure_event.5.a
        default_option = yes
			add_modifier = {
				name = ely_big_money
				months = 60
			}
		random_state = {
			limit = {
				state_region = s:STATE_FLORIDA
			}
			add_modifier = {
				name = modifier_oigos_marshes_drain
				months = -1
			}
		}
	}
	option = {
		name = ely_infrastructure_event.1.b
			add_modifier = {
				name = government_admits_failure
				months = 120
			}
	}
}
