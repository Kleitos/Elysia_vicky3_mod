		if = {
			limit = {
				NOT = { exists = c:MAR }

				tag = MAR

			limit = {
				exists = c:MAR

			limit = {
				NOT = { exists = c:MAR }
			}

				limit = {
					is_secessionist = yes

				change_tag = MAR
			}
		}
		if = {

		}

			add_modifier = {

			activate_law = law_type:law_oligarchy

					birth_date = 1806.9.3 # excact date unknown. 
					culture = cu:marathi
					is_general = yes
					religion = hindu
					interest_group = ig_landowners
					ideology = ideology_royalist
					traits = { 
						resupply_commander 
						basic_political_operator 
					}
				}
				create_character = {
					first_name = "Pratapsinha"
					last_name = "Bhonsle"
					historical = yes
					birth_date = 1793.1.18
					culture = cu:marathi
					ruler = yes
					religion = hindu
					noble = yes
					interest_group = ig_landowners
					ideology = ideology_royalist
					traits = { 
						reserved 
						romantic 
						expensive_tastes
					}
				}
				create_character = { # Mirza Fakhru historically died in 1856, potentially from poisoning.
					first_name = "Shahuji"			
					last_name = "Bhonsle"
					historical = yes
					birth_date = 1802.4.1 # excact date unknown.
