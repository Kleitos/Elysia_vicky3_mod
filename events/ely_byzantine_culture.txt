byzantine_culture.2 = {
	type = country_event
	placement = ROOT
	
	title = byzantine_culture.2.t
	desc = byzantine_culture.2.d
	flavor = byzantine_culture.2.f
	
	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	
	duration = 3
	
	trigger = {
	}

	immediate = {
		cu:bulgarian = {
			save_scope_as = bulgarian_scope
		}
	}

	option = {
		name = byzantine_culture.1.a
		default_option = yes
		ai_chance = {
			base = 100
		}
		remove_primary_culture = cu:bulgarian
			every_country = {
				every_scope_pop = {
					limit = { culture = cu:bulgarian }
					change_pop_culture = { target = cu:greek value = 1 }
				}
			}

		s:STATE_BULGARIA = {
			remove_homeland = cu:bulgarian
			add_homeland = cu:greek
		}
		s:STATE_NORTHERN_THRACE = {
			remove_homeland = cu:bulgarian
		}
		s:STATE_WESTERN_THRACE = {
			remove_homeland = cu:bulgarian
		}
		s:STATE_DOBRUDJA = {
			remove_homeland = cu:bulgarian
			add_homeland = cu:greek
		}
		s:STATE_SKOPIA = {
			remove_homeland = cu:bulgarian
		}
	}
}

byzantine_failure.1 = {
	type = country_event
	placement = ROOT
	
	title = byzantine_failure.1.t
	desc = byzantine_failure.1.d
	flavor = byzantine_failure.1.f
	
	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	
	duration = 3
	
	trigger = {
	}

	immediate = {
		if = {
			limit = {
				NOT = { exists = c:BUL }
				exists = s:STATE_BULGARIA.region_state:BYZ
			}
			create_country = {
				tag = BUL
				origin = c:BYZ
				state = s:STATE_BULGARIA.region_state:BYZ
			}
		}
		else_if = {
			limit = {
				exists = c:BUL
				exists = s:STATE_BULGARIA.region_state:BYZ
			}
			s:STATE_BULGARIA.region_state:BYZ = { set_state_owner = c:BUL }
		}
		else_if = {
			limit = {
				NOT = { exists = c:BUL }
			}
			random_country = {
				limit = {
					is_secessionist = yes
					any_scope_state = {
						state_region = s:STATE_BULGARIA
					}
				}
				change_tag = BUL
			}
		}
		if = {
			limit = { exists = s:STATE_NORTHERN_THRACE.region_state:BYZ }
			s:STATE_NORTHERN_THRACE.region_state:BYZ = { set_state_owner = c:BUL }
		}
		if = {
			limit = { exists = s:STATE_DOBRUDJA.region_state:BYZ }
			s:STATE_DOBRUDJA.region_state:BYZ = { set_state_owner = c:BUL }
		}
		c:BUL = {
			activate_law = law_type:law_presidential_republic
			activate_law = law_type:law_elected_bureaucrats
			activate_law = law_type:law_wealth_voting
		}
		if = {
			limit = {
				c:BYZ ?= this
			}
			remove_primary_culture = cu:bulgarian
		}
	}

	option = {
		name = byzantine_failure.1.a
		default_option = yes
		ai_chance = {
			base = 100
		}
	}

	option = {
		name = byzantine_failure.1.b
		play_as = c:BUL
	}
}
