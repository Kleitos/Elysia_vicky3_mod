namespace = ely_capital
namespace = ely_capital_fix


ely_capital.1 = {
	type = country_event
	placement = root
	title = ely_capital.1.t
	desc = ely_capital.1.d

	duration = 3

	event_image = {
		video = "votp_agitator_speaking"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	trigger = {
		# Triggered by JE
	}

	immediate = {
			add_journal_entry = { type = je_ely_capital_fix }
	}

	option = {
		name = ely_capital.1.a
		add_radicals = {
			value = very_large_radicals
		}	
		add_modifier = {
			name = no_meta_game
			months = -1
		}
	}
	option = {
		name = ely_capital.1.b
		default_option = yes
		add_journal_entry = { type = je_ely_capital }
		ai_chance = {
			base = 100
		}
		set_capital = STATE_DISTRICT_OF_COLUMBIA
	}
	
}
