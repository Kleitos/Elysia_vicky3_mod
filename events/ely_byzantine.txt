byzantine_glory.1 = {
	type = country_event
	placement = ROOT

	title = byzantine_glory.1.t
	desc = byzantine_glory.1.d
	flavor = byzantine_glory.1.f

	event_image = {
		video = "gfx/event_pictures/unspecific_military_parade.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
					state_region = {
						is_homeland = cu:turkish
					}
				}
			save_scope_as = turkish_byzantine_scope
		}
	}

    option = {
		name = byzantine_glory.1.a
		default_option = yes
		add_modifier = {
			name = greek_nationalism
			months = long_modifier_time
		}
		every_scope_state = {
					state_region = {
						is_homeland = cu:turkish
					}
			add_modifier = {
				name = decree_put_the_riot_down
				months = long_modifier_time
			}
		}
	}
}
