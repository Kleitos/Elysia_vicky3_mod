italian_wars.1 = {
	type = country_event
	placement = ROOT

	title = italian_wars.1.t
	desc = italian_wars.1.d
	flavor = italian_wars.1.f

	event_image = {
		video = "gfx/event_pictures/unspecific_politicians_arguing.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	trigger = {
	}

	immediate = {	
		c:LMV = {
			annex = c:SIC
			trigger_event = { id = italian_wars.3 days = 730 }
		}
		c:BYZ = {
			trigger_event = { id = italian_wars.2 }
		}
	}

    option = {
		name = italian_wars.1.a
		default_option = yes
		create_diplomatic_play = {
			target_country = c:SAR
			war = no
			escalation = 81
			type = dp_annex_war
			target_state = s:STATE_PIEDMONT.region_state:SAR
			add_war_goal = {
				holder = c:SAR
				type = annex_country
				target_state = s:STATE_LOMBARDY.region_state:LMV
			}
		}
	}
}
