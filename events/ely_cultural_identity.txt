elyric.1 = {
	type = country_event
	placement = ROOT

	title = elyric.1.t
	desc = elyric.1.d
	flavor = elyric.1.f

	duration = 3

	event_image = {
		video = "europenorthamerica_native_american"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	minor_left_icon = c:ELY
	#minor_right_icon = scope:integrate_elyric_target_country

	immediate = {
		random_neighbouring_state = {
			limit = {
				owner = {
					is_subject_of = ROOT
					country_has_primary_culture = cu:elyric
				}
			}
			owner = {
				save_scope_as = integrate_elyric_target_state_1
			}
		}
		random_neighbouring_state = {
			limit = {
				owner = {
					is_subject_of = ROOT
					country_has_primary_culture = cu:elyric
				}
				NOT = { this = scope:integrate_elyric_target_state_1 }
			}
			owner = {
				save_scope_as = integrate_elyric_target_state_2
			}
		}
		random_neighbouring_state = {
			limit = {
				owner = {
					is_subject_of = ROOT
					country_has_primary_culture = cu:elyric
				}
				NOT = { this = scope:integrate_elyric_target_state_1 }
				NOT = { this = scope:integrate_elyric_target_state_2 }
			}
			owner = {
				save_scope_as = integrate_elyric_target_state_3
			}
		}
	}

	option = {
		name = elyric.1.a
		default_option = yes
		annex = scope:integrate_elyric_target_state_1
		ai_chance = {
			base = 1
		}
	}

	option = {
		name = elyric.1.b
		trigger = {
			exists = scope:integrate_elyric_target_state_2
		}
		annex = scope:integrate_elyric_target_state_2
	}
	
	option = {
		name = elyric.1.c
		trigger = {
			exists = scope:integrate_elyric_target_state_3
		}
		annex = scope:integrate_elyric_target_state_3
	}
}
