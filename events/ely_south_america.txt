### Cultural Shift ###

ely_south_america_integration.1 = {
	type = country_event
	placement = ROOT
	
	title = ely_south_america_integration.1.t
	desc = ely_south_america_integration.1.d
	flavor = ely_south_america_integration.1.f

	event_image = {
		video = "gfx/event_pictures/southamerica_aristocrats.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	
	duration = 3
	
	trigger = {
	}

	immediate = {
		if = {
			limit = {
				c:TWT ?= this
			}
			remove_primary_culture = cu:aimara
		}
		if = {
			limit = {
				c:INC ?= this
			}
			remove_primary_culture = cu:aimara
		}
	}

	option = {
		name = cultural_shift.1.a
		default_option = yes
		ai_chance = {
			base = 100
		}
		trigger_event = { id = ely_south_america_integration.2 days = 10 }	
	}
}


ely_south_america_integration.2 = {
	type = country_event
	placement = ROOT
	
	title = ely_south_america_integration.2.t
	desc = ely_south_america_integration.2.d
	flavor = ely_south_america_integration.2.f
	
	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	
	duration = 3
	
	trigger = {
	}

	immediate = {
	}

	option = {
		name = ely_south_america_integration.2.a
		default_option = yes
		ai_chance = {
			base = 100
		}
			every_country = {
				every_scope_pop = {
					limit = { culture = cu:aimara }
					change_pop_culture = { target = cu:quechua value = 1 }
				}
			}
			
		s:STATE_AREQUIPA = {
			remove_homeland = cu:aimara
		}
		s:STATE_LA_PAZ = {
			remove_homeland = cu:aimara
		}
		s:STATE_POTOSI = {
			remove_homeland = cu:aimara
		}
		s:STATE_TARAPACA = {
			remove_homeland = cu:aimara
		}
		s:STATE_ANTOFAGASTA = {
			remove_homeland = cu:aimara
		}
		s:STATE_SANTIAGO = {
			remove_homeland = cu:aimara
			add_homeland = cu:quechua
		}
		s:STATE_LOS_RIOS = {
			remove_homeland = cu:aimara
			add_homeland = cu:quechua
		}
		s:STATE_ARAUCANIA = {
			remove_homeland = cu:aimara
			add_homeland = cu:quechua
		}
	}
}
