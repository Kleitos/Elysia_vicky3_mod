ely_south_america_diplo.1 = {
	type = country_event

	title = ely_south_america_diplo.1.t
	desc = ely_south_america_diplo.1.d
	flavor = ely_south_america_diplo.1.f

	event_image = {
		video = "gfx/event_pictures/southamerica_aristocrats.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 4

	trigger = {
	}

	immediate = {
		c:ELY = {
			remove_diplomatic_pact = {
				country = c:MUI
				type = puppet
			}
			create_bidirectional_truce = {
				country = c:TWT
				months = 40
			}
		}
		c:MUI = {
			set_variable = { name = inc_imperial_target }
		}
	}
	option = {
		name = ely_south_america_diplo.1.a 
		default_option = yes
			add_journal_entry = {
				type = je_muiscan_unification
			}
			
		c:TWT = {
			create_diplomatic_pact = {
				country = c:MUI
				type = puppet
			}
		}
	}
}

### Cultural Shift ###

ely_south_america_integration.1 = {
	type = country_event
	placement = ROOT
	
	title = ely_south_america_integration.1.t
	desc = ely_south_america_integration.1.d
	flavor = ely_south_america_integration.1.f

	event_image = {
		video = "gfx/event_pictures/southamerica_aristocrats.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	
	duration = 3
	
	trigger = {
	}

	immediate = {
	}

	option = {
		name = cultural_shift.1.a
		default_option = yes
		ai_chance = {
			base = 100
		}
		c:TWT = {
			trigger_event = { id = ely_south_america_integration.2 }
		}
		remove_primary_culture = cu:aimara
		remove_primary_culture = cu:quechua
		add_primary_culture = cu:incan_culture
		hidden_effect = {
			every_country = {
				every_scope_pop = {
					limit = { culture = cu:aimara }
					change_pop_culture = { target = cu:incan_culture value = 1 }
				}
			}
				every_scope_pop = {
					limit = { culture = cu:quechua }
