franco_german_alliance = {
	is_shown = {
 		is_player = yes
		exists = c:FRA
		exists = c:GER
		NOT = {
			has_global_variable = break_franco_germany
		}
	}

	possible = {
		OR = {	
			c:FRA = {
				NOT = {	
					has_diplomatic_pact = {
						type = alliance
						who = c:GER
					}
				}	
			}
			c:GER= {
				NOT = {	
					has_diplomatic_pact = {
						type = alliance
						who = c:FRA
					}
				}	
			}
		}
	}

	when_taken = {
		trigger_event = {
			id = ely_alliance.1
			days = 0
			popup = yes
		}
	}
}
