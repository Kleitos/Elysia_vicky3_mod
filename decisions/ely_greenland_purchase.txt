ely_purchase_greenland = {
	is_shown = {
		exists = c:ELY
		exists = c:DEN
		c:ELY = ROOT
		is_subject = no
		NOT = {
			has_variable = greenland_purchase_var
		}
		c:DEN = {
			owns_entire_state_region = STATE_GREENLAND
			is_player = no
		}
	}

	possible = {
		owns_entire_state_region = STATE_WASHINGTON
		in_default = no
		taking_loans = no
		is_at_war = no
		root.gdp > 60000000
		c:DEN = {
			relations:root >= relations_threshold:cordial
			is_at_war = no
			OR = {
				in_default = yes
				taking_loans = yes
			}
		}
		custom_tooltip = {
			text = greenlandinprog_tt
			NOT = {
				has_global_variable = greenland_purchase_in_prog
			}
		}
	}

	when_taken = {
		set_variable = {
			name = greenland_purchase_var
			value = yes
		}
		set_global_variable = {
			name = greenland_purchase_in_prog
		}
		trigger_event = {
			id = greenland.4
			days = 0
			popup = yes
		}
	}

	ai_chance = {
		base = 10	
		modifier = {
			trigger = { gold_reserves > 3000000 }
			add = 10
		}		
		modifier = {
			trigger = { 
				gold_reserves > 0
				weekly_net_fixed_income > 100000
			}
			add = 10
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}
		modifier = {
			c:DEN = {
				is_subject_of = c:SWE
			}
			add = 1000
		}
	}
}
