integrate_elyric = {
	is_shown = {
		exists = c:ELY
		root = c:ELY
		NOT = { has_variable = integrate_elyric_ended }
		has_variable = integrate_elyric_started
	}

	possible = {
		is_at_war = no
		custom_tooltip = {
			text = integrate_elyric_cooldown_tt
			NOT = { has_variable = integrate_elyric_cooldown }
		}
		any_subject_or_below = {
			country_has_primary_culture = cu:elyric
			relations:root >= relations_threshold:amicable
		}
	}

	when_taken = {
		trigger_event = { id = elyric.1 days = 0 popup = yes }
		set_variable = {
			name = integrate_elyric_cooldown
			years = 1
		}
	}

	ai_chance = {
		value = 100
	}
}
