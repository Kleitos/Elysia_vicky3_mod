formalis_elysian_games_decision = {
	is_shown = {
		exists = c:ELY
		this = c:ELY
	}

	possible = {
		this = c:ELY
		has_technology_researched = organized_sports
		hidden_trigger = {
			NOT = { has_variable = formalis_elysian_games_decision_taken }
		}
	}
	
	when_taken = {
		set_variable = {
			name = formalis_elysian_games_decision_taken
			value = yes
		}
		add_modifier = {
			name = modifier_elysian_games
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = { bureaucracy > 0 }
			add = 5
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}		
	}
}
