je_integrate_slovenes = {
	icon = "gfx/interface/icons/event_icons/unify.dds"

	group = je_group_elysian_content

	complete = {
		cu:slovene = {
			NOT = { culture_is_discriminated_in = ROOT }
		}
		owns_entire_state_region = STATE_ISTRIA
		owns_entire_state_region = STATE_SLOVENIA
				NOT = {
					any_scope_state = {
						state_region = {
							is_homeland = cu:slovene
						}
						radical_fraction = {
							value >= 0.10
							culture = cu:slovene
						}
					}
				}
		any_scope_state = {
			state_region = s:STATE_SLOVENIA
			is_incorporated = yes
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 2
			}
		}
		any_scope_state = {
			state_region = s:STATE_ISTRIA
			is_incorporated = yes
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 2
			}
		}
	}

	on_complete = {
		change_variable = {
			name = danubian_fed_var
			add = 1
		}
		set_variable = {
			name = slovenes_integrated
		}
		trigger_event = { id = danubian_culture.1 }
	}

	fail = {
		any_scope_state = {
			state_region = {
				is_homeland = cu:slovene
			}
			radical_fraction = {
				value >= 0.5
				culture = cu:slovene
			}
		}
	}

	on_fail = {
		trigger_event = { id = danubian_breakup.1 }
	}
}
