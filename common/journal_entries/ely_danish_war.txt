je_danish_war_GBR = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_elysian_content

	complete = {
		owns_entire_state_region = STATE_ZANZIBAR
		OR = {
			NOT = { exists = c:SEI }
			c:SEI ?= {
				is_country_alive = no
			}
		}
		OR = {
			NOT = { exists = c:SAS }
			c:SAS ?= {
				is_country_alive = no
			}
		}
		OR = {
			NOT = { exists = c:WAS }
			c:WAS ?= {
				is_country_alive = no
			}
		}
		OR = {
			NOT = { exists = c:SIL }
			c:SIL ?= {
				is_country_alive = no
			}
		}
		has_truce_with = c:SPA
		has_truce_with = c:SWE
	}

	on_complete = {
		trigger_event = { id = danish_english_victory.1 popup = yes }
	}

	fail = {
		OR = {
			exists = c:SEI
			c:SEI ?= {
				is_subject_of = c:SPA
				is_country_alive = yes
				has_truce_with = c:GBR
			}
		}
		OR = {
			exists = c:SAS 
			c:SAS ?= {
				is_subject_of = c:SWE
				is_country_alive = yes
				has_truce_with = c:GBR
			}
		}
		OR = {
			exists = c:WAS 
			c:WAS ?= {
				is_subject_of = c:SWE
				is_country_alive = yes
				has_truce_with = c:GBR
			}
		}
		OR = {
			exists = c:SIL 
			c:SIL ?= {
				is_subject_of = c:SPA
				is_country_alive = yes
				has_truce_with = c:GBR
			}
		}
		c:SPA ?= {
			owns_entire_state_region = STATE_ZANZIBAR
			has_truce_with = c:GBR
		}
		c:SWE ?= {
			has_truce_with = c:GBR
		}
	}

	on_fail = {
		trigger_event = { id = danish_english_defeat.1 popup = yes }
	}

	weight = 10

	should_be_pinned_by_default = yes

	timeout = 1825

	on_timeout = {
		trigger_event = { id = danish_war_inconclusive.1 popup = yes }
	}
	
}
