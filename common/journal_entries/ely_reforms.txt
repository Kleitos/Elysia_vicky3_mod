je_reform_bureaucracy = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_elysian_content

	is_shown_when_inactive = {
		has_modifier = outmoded_bureaucracy
		NOT = { c:ELY ?= THIS }
		NOT = { this = c:TUR }
	}

	possible = {
		has_modifier = outmoded_bureaucracy
	}

	complete = {
		NOR = { 
			has_law = law_type:law_hereditary_bureaucrats 
			has_law = law_type:law_land_based_taxation
			has_law = law_type:law_consumption_based_taxation
		}
		bureaucracy >= 0 
		approaching_bureaucracy_shortage = no
		has_technology_researched = identification_documents
	}

	on_complete = {
		remove_modifier = outmoded_bureaucracy
	}
}

je_reform_bureaucracy_ely = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_elysian_content

	is_shown_when_inactive = {
		c:ELY ?= THIS
	}

	possible = {
		c:ELY ?= THIS
	}

	complete = {
		NOR = { 
			has_law = law_type:law_hereditary_bureaucrats 
			has_law = law_type:law_land_based_taxation
			has_law = law_type:law_consumption_based_taxation
		}
		bureaucracy >= 1000 
		approaching_bureaucracy_shortage = no
		has_technology_researched = identification_documents
	}

	on_complete = {
		if = {
			limit = {
				exists = c:ELY
				c:ELY = ROOT
			}
		trigger_event = { id = ely_reform_event.1 }
		}
	}

	timeout = 18262

	on_timeout = {
		trigger_event = { id = ely_reform_event.3 popup = yes }
	}
}

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_elysian_content

	is_shown_when_inactive = {
		has_modifier = religious_fundamentalism
	}

	possible = {		
		has_modifier = religious_fundamentalism
	}

	complete = {
		NOT = { has_law = law_type:law_theocracy }
		OR = { 
			has_law = law_type:law_right_of_assembly
			has_law = law_type:law_protected_speech
		}
		OR = { 
			has_law = law_type:law_graduated_taxation
			has_law = law_type:law_proportional_taxation
			has_law = law_type:law_per_capita_based_taxation
		}
		NOR  = { 
			has_law = law_type:law_traditionalism
			has_law = law_type:law_industry_banned
		}
		OR = { 
			has_law = law_type:law_homesteading
			has_law = law_type:law_collectivized_agriculture
			has_law = law_type:law_commercialized_agriculture
		}
		NOR = { 
			has_law = law_type:law_autocracy
			has_law = law_type:law_oligarchy
		}
		NOT = { has_law = law_type:law_state_religion }
		has_law = law_type:law_slavery_banned 
		AND = {
			ig:ig_landowners = {
				is_powerful = no
			}
			ig:ig_devout = {
				is_powerful = no
			}
		}
	}

	on_complete = {
		if = {
			limit = {
				has_modifier = religious_fundamentalism
			}
			remove_modifier = religious_fundamentalism
		}
	}
}
