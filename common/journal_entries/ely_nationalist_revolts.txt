je_avert_italian_mutiny = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_elysian_content

	possible = {
		owns_entire_state_region = STATE_ISTRIA
	}

	complete = {
		NOT = { has_variable = italian_revolt_event }
		cu:north_italian = {
			NOT = { culture_is_discriminated_in = ROOT }
		}
		
	}

	on_complete = {
	
	}

	fail = {
		any_scope_state = {
			state_region = {
				is_homeland = cu:north_italian
			}
			radical_fraction = {
				value >= 0.15
				culture = cu:north_italian
			}
		}
	}

	on_fail = {
			add_modifier = {
				name = weakened_by_mutiny
				months = normal_modifier_time
			}
		trigger_event = { id = italian_revolt.1 }
	}

}
