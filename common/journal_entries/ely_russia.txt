je_russian_civil_war_NOV = {
    icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_elysian_content

	complete = {
 		is_at_war = no
		NOT = { has_truce_with = c:RUS }
		is_subject = no
		OR = {
			NOT = { exists = c:RUS }
			c:RUS ?= {
				is_country_alive = no
			}
		}
	}

	on_complete = {
		if = {
			limit = {
				exists = c:NOV
				c:NOV = ROOT
			}
		trigger_event = { id = russian_revolution.3 }
		}
	}
}
