je_guarani_uprising_ARG = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_elysian_content

	complete = {
		c:ARG ?= {
			is_country_alive = yes
			has_modifier = danish_war_exhausted
		}
	}

	on_complete = {
		if = {
			limit = {
				exists = c:GRI
				c:GRI = ROOT
			}
			c:GRI = {
				trigger_event = { id = native_guarani.1 popup = yes }
			}
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes

	timeout = 3650
}
