je_mandika_uprising_MLI = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_elysian_content

	complete = {
		AND = {
			c:SIL ?= {
				is_country_alive = yes
				has_modifier = danish_war_exhausted
			}
			c:LIB ?= {
				is_country_alive = yes
				has_modifier = danish_war_exhausted
			}
		}
	}

	on_complete = {
		if = {
			limit = {
				exists = c:KBU
				c:KBU = ROOT
			}
			c:KBU = {
				trigger_event = { id = native_mandika.1 popup = yes }
			}
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes

	timeout = 3650
}
