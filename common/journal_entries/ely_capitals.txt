je_ely_capital = {
	icon = "gfx/interface/icons/event_icons/unify.dds"

	group = je_group_elysian_content

	is_shown_when_inactive = {
		exists = c:ELY
		THIS = c:ELY
	}

	possible = {
		THIS = c:ELY
	}

	complete = {
		owns_entire_state_region = STATE_DISTRICT_OF_COLUMBIA
		any_scope_state = {
			state_region = s:STATE_DISTRICT_OF_COLUMBIA
				is_capital = no
		}
	}

	on_complete = {
		trigger_event = { id = ely_capital.1 }
	}
}
