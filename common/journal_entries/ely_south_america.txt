je_muiscan_transfer = {
	icon = "gfx/interface/icons/event_icons/event_election.dds"

	group = je_group_elysian_content

	is_shown_when_inactive = {
		c:TWT ?= THIS
	}

	possible = {
		c:TWT ?= THIS
		NOT = { has_truce_with = c:ELY }
	}

	complete = {
		country_rank >= rank_value:major_power
		c:ELY = { relations:root >= relations_threshold:amicable }
		army_size >= 100
	}

	on_complete = {
		if = {
			limit = {
				exists = c:TWT
				c:TWT = ROOT
			}
		trigger_event = { id = ely_south_america_diplo.1 }
		}
	}
}

je_cultural_shift_incan = {
	icon = "gfx/interface/icons/event_icons/unify.dds"

	group = je_group_elysian_content

	is_shown_when_inactive = {
		c:TWT ?= THIS
	}

	possible = {
		c:TWT ?= THIS
	}

	complete = {
	OR = { 
		AND = {	
            is_ai = no
				has_law = law_type:law_compulsory_primary_school
				OR = {
					has_law = law_type:law_religious_schools
					has_law = law_type:law_private_schools
					has_law = law_type:law_public_schools
				}
				institution_investment_level = {
					institution = institution_schools
					value >= 3
				}
				institution_investment_level = {
					institution = institution_police
					value >= 2
				}
				institution_investment_level = {
					institution = institution_home_affairs
					value >= 1
				}
			}
		AND = {
				is_ai = yes
				year >= 1886
			}	
		}
	}

	on_complete = {
		if = {
			limit = {
				exists = c:TWT
				c:TWT = ROOT
			}
		trigger_event = { id = ely_south_america_integration.1 }
		}
	}

}
