je_dual_monarchy = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	group = je_group_historical_content

	complete = {
		cu:hungarian = {
			NOT = { culture_is_discriminated_in = ROOT }
		}
		owns_entire_state_region = STATE_BANAT
		owns_entire_state_region = STATE_SOUTHERN_TRANSYLVANIA
		owns_entire_state_region = STATE_NORTHERN_TRANSYLVANIA
		owns_entire_state_region = STATE_BEKES
		owns_entire_state_region = STATE_TRANSDANUBIA
		owns_entire_state_region = STATE_CENTRAL_HUNGARY
		owns_entire_state_region = STATE_DELVIDEK
				NOT = {
					any_scope_state = {
						state_region = {
							is_homeland = cu:hungarian
						}
						radical_fraction = {
							value >= 0.10
							culture = cu:hungarian
						}
					}
				}
		any_scope_state = {
			state_region = s:STATE_CENTRAL_HUNGARY
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 2
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = dual_monarchy.1
		}
		change_variable = {
			name = danubian_fed_var
			add = 1
		}
	}

	fail = {
		cu:hungarian = {
			culture_secession_progress = {
				target = ROOT
				value > 0.5
			}
		}
	}

	on_fail = {
		set_variable = dual_monarchy_fail
		trigger_event = {
			id = dual_monarchy.1
		}
	}

	invalid = {
		OR = {
			custom_tooltip = {
				text = austria_hunary_invalid
				NOT = {
					any_scope_state = {
						state_region = {
							is_homeland = cu:hungarian
						}
					}
				}
			}
			NOT = {
				this = c:AUS
			}
		}
	}

	weight = 100
	should_be_pinned_by_default = yes
}
