je_aboriginal_uprising_AUS = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_elysian_content

	complete = {
		AND = {
			OR = {
				c:NSW ?= {
					is_country_alive = yes
					has_modifier = danish_war_exhausted
				}
				c:AST ?= {
					is_country_alive = yes
					has_modifier = danish_war_exhausted
				}
			}
			c:WAS ?= {
				is_country_alive = yes
				has_modifier = danish_war_exhausted
			}
			c:SAS ?= {
				is_country_alive = yes
				has_modifier = danish_war_exhausted
			}
		}
	}

	on_complete = {
		if = {
			limit = {
				exists = c:KAU
				c:KAU = ROOT
			}
			c:KAU = {
				trigger_event = { id = native_australia.1 popup = yes }
			}
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes

	timeout = 3650
}
