je_solomonid_main = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	
	group = je_group_historical_content

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		trigger_event = {
			id = westernization.2
		}
	}

	on_monthly_pulse = {
		random_events = {
			100 = 0
			10 = westernization.4
			10 = westernization.5
			10 = westernization.6
		}
	}

	current_value = {
		value = var:solomonid_var
	}

	goal_add_value = {
		add = 3
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}

je_solomonid_economy = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	
	group = je_group_historical_content

	complete = {
		in_default = no
		any_scope_state = {
			filter = {
				is_incorporated = yes
			}
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 5
			}
			has_building = building_railway
			percent > 0.7
		}
	}

	on_complete = {
		change_variable = {
			name = solomonid_var
			add = 1
		}
	}

	on_monthly_pulse = {
		random_events = {
			100 = 0
			10 = westernization.7
			10 = westernization.8
		}
	}

	transferable = no

	weight = 5000
}
