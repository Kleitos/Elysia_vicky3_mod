je_byzantine_resurgence = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_elysian_content

	complete = {
 		is_at_war = no
		country_rank >= rank_value:minor_power
		has_law = law_type:law_professional_army
		has_technology_researched = general_staff
		c:BYZ = {
			army_size > { 
    		value =  c:TUR.army_size
    		multiply = 1
			}
		}
		NOT = { has_truce_with = c:TUR }
		is_subject = no
		owns_entire_state_region = STATE_IONIAN_ISLANDS
		owns_entire_state_region = STATE_CRETE
		owns_entire_state_region = STATE_WEST_AEGEAN_ISLANDS
	}

	on_complete = {
		if = {
			limit = {
				exists = c:BYZ
				c:BYZ = ROOT
			}
		trigger_event = { id = byzantine_resurgence.1 }
		}
	}
}

je_cultural_shift_byz_bulgarian = {
	icon = "gfx/interface/icons/event_icons/unify.dds"

	group = je_group_elysian_content

	complete = {
	OR = { 
		AND = {	
            is_ai = no
		owns_entire_state_region = STATE_DOBRUDJA
		owns_entire_state_region = STATE_SKOPIA
		owns_entire_state_region = STATE_WESTERN_THRACE
		owns_entire_state_region = STATE_NORTHERN_THRACE
		owns_entire_state_region = STATE_BULGARIA
		OR = {
			has_law = law_type:law_religious_schools
			has_law = law_type:law_private_schools
			has_law = law_type:law_public_schools
		}
		institution_investment_level = {
			institution = institution_schools
			value >= 1
		}
		institution_investment_level = {
			institution = institution_police
			value >= 1
		}
		institution_investment_level = {
			institution = institution_home_affairs
			value >= 1
		}
			}
		AND = {
				is_ai = yes
				year >= 1842
			}	
		}
	}
	

	on_complete = {
		change_variable = {
			name = rhoman_var
			add = 1
		}
		if = {
			limit = {
				exists = c:BYZ
				c:BYZ = ROOT
			}
		trigger_event = { id = byzantine_culture.2 }
		}
	}

	fail = {
		any_scope_state = {
			state_region = {
				is_homeland = cu:bulgarian
			}
			radical_fraction = {
				value >= 0.5
				culture = cu:bulgarian
			}
		}
	}

	on_fail = {
		trigger_event = { id = byzantine_failure.1 }
	}

}
